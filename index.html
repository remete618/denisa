package at.spengergasse.sj2526pos1javaseed.presentation.api;

import org.springframework.http.HttpStatus;
import org.springframework.http.ProblemDetail;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.ControllerAdvice;
import org.springframework.web.bind.annotation.ExceptionHandler;

@ControllerAdvice
public class GlobalControllerAdvice {

    @ExceptionHandler
    public ResponseEntity<ProblemDetail> handleException(Throwable t) {
        ProblemDetail problemDetail = ProblemDetail.forStatus(HttpStatus.INTERNAL_SERVER_ERROR);
        problemDetail.setDetail(t.getMessage());
        return ResponseEntity.internalServerError().body(problemDetail);
    }

}

What changed: The error detail now shows the actual exception message instead of a hardcoded joke string.

Step 3: Add Dependencies to pom.xml

Add Two Dependencies

File: pom.xml

Open pom.xml and find the <dependencies> section. Add these two blocks before the closing </dependencies> tag:

3a. HATEOAS (for hypermedia links in API responses)

Find this line in your pom.xml:

<dependency>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-starter-web</artifactId>
</dependency>

Add right after it:

<dependency>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-starter-hateoas</artifactId>
</dependency>

3b. Spring REST Docs MockMVC (for generating documentation from tests)

Find this line in your pom.xml:

<dependency>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-starter-test</artifactId>
    <scope>test</scope>
</dependency>

Add right after it:

<dependency>
    <groupId>org.springframework.restdocs</groupId>
    <artifactId>spring-restdocs-mockmvc</artifactId>
    <scope>test</scope>
</dependency>
After this step
Click "Load Maven Changes" in IntelliJ (blue icon top-right) to download the new dependencies.

Step 4: Add Maven Plugins to pom.xml

Add Two Plugins

File: pom.xml

Find the <plugins> section inside <build>. After the existing spring-boot-maven-plugin block, add these two plugins:

4a. AsciiDoctor Maven Plugin (converts docs to HTML)

<plugin>
    <groupId>org.asciidoctor</groupId>
    <artifactId>asciidoctor-maven-plugin</artifactId>
    <version>2.2.6</version>
    <executions>
        <execution>
            <id>generate-docs</id>
            <phase>prepare-package</phase>
            <goals>
                <goal>process-asciidoc</goal>
            </goals>
            <configuration>
                <backend>html</backend>
                <doctype>book</doctype>
            </configuration>
        </execution>
    </executions>
    <dependencies>
        <dependency>
            <groupId>org.springframework.restdocs</groupId>
            <artifactId>spring-restdocs-asciidoctor</artifactId>
            <version>${spring-restdocs.version}</version>
        </dependency>
    </dependencies>
</plugin>

4b. Maven Resources Plugin (copies docs to static folder)

<plugin>
    <groupId>org.apache.maven.plugins</groupId>
    <artifactId>maven-resources-plugin</artifactId>
    <executions>
        <execution>
            <id>copy-docs</id>
            <phase>prepare-package</phase>
            <goals>
                <goal>copy-resources</goal>
            </goals>
            <configuration>
                <outputDirectory>${project.build.outputDirectory}/static/docs</outputDirectory>
                <resources>
                    <resource>
                        <directory>${project.build.directory}/generated-docs</directory>
                    </resource>
                </resources>
            </configuration>
        </execution>
    </executions>
</plugin>

How it works: When you run mvn package, AsciiDoctor converts your .adoc file to HTML, and the resources plugin copies it into the JAR under /static/docs so it's accessible via the web server.

package at.spengergasse.sj2526pos1javaseed.presentation.api;

import at.spengergasse.sj2526pos1javaseed.commands.CreateGameCommand;
import at.spengergasse.sj2526pos1javaseed.domain.Game;
import at.spengergasse.sj2526pos1javaseed.dtos.GameDto;
import at.spengergasse.sj2526pos1javaseed.service.GameService;
import jakarta.validation.Valid;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.data.domain.Pageable;
import org.springframework.data.domain.Slice;
import org.springframework.data.web.PageableDefault;
import org.springframework.hateoas.EntityModel;
import org.springframework.http.ResponseEntity;
import org.springframework.validation.BindingResult;
import org.springframework.web.bind.annotation.*;

import java.net.URI;

import static org.springframework.hateoas.server.mvc.WebMvcLinkBuilder.*;

@RequiredArgsConstructor
@Slf4j
@RestController
@RequestMapping("/api/games")
public class GameController {

    private final GameService gameService;

    @GetMapping
    public ResponseEntity<Slice<GameDto>> getGames(@PageableDefault(size = 10) Pageable pageable) {
        log.info("GET /api/games");
        var games = gameService.getGames(pageable);
        log.info("GET /api/games -> {} games", games.getNumberOfElements());
        return (games.isEmpty()) ? ResponseEntity.noContent().build()
                : ResponseEntity.ok(games);
    }

    @GetMapping("/{key}")
    public ResponseEntity<EntityModel<GameDto>> getReview(@PathVariable Long key,
                                                           @PageableDefault(size = 1) Pageable pageable) {
        log.info("GET /api/games/{}", key);
        var game = gameService.getGame(key, pageable);
        if (game == null) {
            return ResponseEntity.noContent().build();
        }
        EntityModel<GameDto> model = EntityModel.of(game,
                linkTo(methodOn(GameController.class).getReview(key, pageable)).withSelfRel(),
                linkTo(methodOn(GameController.class).getGames(pageable)).withRel("games"));
        return ResponseEntity.ok(model);
    }

    @PostMapping
    public ResponseEntity<GameDto> createReview(@RequestBody @Valid CreateGameCommand cmd,
                                                 BindingResult bindingResult) {
        log.info("POST /api/games - create game");
        if (bindingResult.hasErrors()) { return ResponseEntity.badRequest().build(); }
        var created = gameService.create(cmd);
        return ResponseEntity.created(URI.create("/api/games/" + created.getId().id()))
                .body(new GameDto(created));
    }

    @DeleteMapping("/{key}")
    public ResponseEntity<Void> deleteGame(@PathVariable Long key) {
        log.info("DELETE /api/games/{}", key);
        return gameService.delete(new Game.GameId(key))
                ? ResponseEntity.noContent().build()
                : ResponseEntity.notFound().build();
    }

}

What changed:

  • The GET single game endpoint now returns EntityModel<GameDto> with HATEOAS links (self link + collection link)
  • Added imports for EntityModel and WebMvcLinkBuilder
  • POST now uses the actual created entity ID in the Location header (was hardcoded to 1)
What HATEOAS looks like in the response

When you GET /api/games/1, the response now includes:

{
  "name": "Zelda",
  "description": "Adventure game",
  "price": 60,
  "_links": {
    "self": { "href": "http://localhost:8080/api/games/1" },
    "games": { "href": "http://localhost:8080/api/games" }
  }
}
